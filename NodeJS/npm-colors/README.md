1.Crear una carpeta llamada colors
2.Crear un archivo "main.js"
# npm desde CLI
3.Iniciamos un proyecto desde npm con el comando
"sh
npm init -y"
4.Buscamos el package colors y los instalamos con el comando
"sh
npm i colors"

"javascript 
const colors = requiere("colors");
